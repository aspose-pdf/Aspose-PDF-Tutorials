//ExSummaryΟ κώδικας αυτός δείχνει πώς να δημιουργηθεί ένα έγγραφο PDF, να προστεθεί μια εικόνα σε αυτό, και να προσαρμοστεί το μέγεθος της εικόνας και τις ιδιότητες της σελίδας.
//ExStepSummary:0: Αυτός ο κώδικας αρχικά αρχίζει το αντικείμενο του εγγράφου και προσθέτει μια νέα σελίδα σε αυτό.
//ExStepSummary:1: Αυτή η ενότητα δημιουργεί μια ενδείξεις εικόνας και καθορίζει τις διαστάσεις και τον τύπο αρχείου.
//ExStepSummary:2: Αυτό το βήμα προσδιορίζει το αρχείο προέλευσης της εικόνας και το προσθέτει στη σελίδα.
//ExStepSummary:3: Αυτό το βήμα καθορίζει τις ιδιότητες της σελίδας, συμπεριλαμβανομένου του πλάτους και του ύψους.
//ExStepSummary:4: Αυτή η ενότητα αποθηκεύει το έγγραφο στην καθορισμένη διαδρομή αρχείου και επαληθεύει τη λειτουργία αποθήκευσης.
//ExStepImage:2:images/1.png
//ExStepImage:3:images/2.png

//ExStep:0-
// Ειδίκεστε τη διαδρομή του καταλόγου εγγράφων
string dataDir = "Your Document Directory";

//ExStep:1-
// Εντοπικό αντικείμενο εγγράφου
Document doc = new Document();
// Προσθήκη νέας σελίδας στο έγγραφο PDF
Aspose.Pdf.Page page = doc.Pages.Add();

//ExStep:2-
// Δημιουργήστε μια περίπτωση εικόνας
Aspose.Pdf.Image img = new Aspose.Pdf.Image();
// Εγκαταστήστε πλάτος εικόνας σε 100 σημεία
img.FixWidth = 100;
// Εγκαταστήστε το ύψος της εικόνας σε 100 σημεία
img.FixHeight = 100;
// Εγκαταστήστε τον τύπο εικόνας ως Άγνωστο (θα υποτεθεί αργότερα)
img.FileType = Aspose.Pdf.ImageFileType.Unknown;
// Ειδίκεστε τη διαδρομή για το αρχείο εικόνας προέλευσης
img.File = dataDir + "aspose-logo.jpg";
// Προσθέστε την εικόνα στις παραγράφους της σελίδας
page.Paragraphs.Add(img);

//ExStep:3-
// Εναρμηνεύστε το πλάτος της σελίδας σε 800 σημεία
page.PageInfo.Width = 800;
// Εναρμηνεύστε το ύψος της σελίδας σε 800 σημεία
page.PageInfo.Height = 800;

//ExStep:4-
// Ειδίκεστε τη διαδρομή αρχείου εξόδου
dataDir = dataDir + "SetImageSize_out.pdf";
// Αποθηκεύστε το έγγραφο PDF που προκύπτει
doc.Save(dataDir);

//ExStep:5-
// Ενημερώστε τον χρήστη για την επιτυχημένη επιχείρηση
Console.WriteLine("\nImage size added successfully.\nFile saved at " + dataDir);
//ExEnd
