## Full Working Example

```csharp
using System;
using System.IO;
using Aspose.Pdf;
using Aspose.Pdf.Facades;

class Program
{
    static void Main()
    {
        // -------------------------------------------------
        // 1я╕ПтГг Load the PDF document
        // -------------------------------------------------
        string sourcePath = Path.Combine(Environment.CurrentDirectory, "YOUR_DIRECTORY", "SignedDoc.pdf");
        Document pdfDocument = new Document(sourcePath);
        Console.WriteLine($"тЬЕ Loaded PDF: {sourcePath}");

        // -------------------------------------------------
        // 2я╕ПтГг List (and optionally extract) signatures
        // -------------------------------------------------
        PdfFileSignature signatureHandler = new PdfFileSignature(pdfDocument);
        string[] signatureNames = signatureHandler.GetSignNames();

        if (signatureNames.Length == 0)
        {
            Console.WriteLine("тЪая╕П No signature fields found.");
        }
        else
        {
            Console.WriteLine("ЁЯЦЛя╕П Signature fields:");
            foreach (var name in signatureNames)
                Console.WriteLine($"- {name}");

            byte[] rawSig = signatureHandler.ExtractSignature(signatureNames[0]);
            string sigOut = Path.Combine(Environment.CurrentDirectory, "YOUR_DIRECTORY", "FirstSignature.bin");
            File.WriteAllBytes(sigOut, rawSig);
            Console.WriteLine($"ЁЯУБ First signature saved to {sigOut}");
        }

        // 3я╕ПтГг Set up PDF/XтАС4 conversion options
        // -------------------------------------------------

            PdfFormat.PDF_X_4,
            ConvertErrorAction.Delete);

        // -------------------------------------------------
        // 4я╕ПтГг Convert and save as PDF/XтАС4
        // -------------------------------------------------
        pdfDocument.Convert(conversionOptions);
        string outputPath = Path.Combine(Environment.CurrentDirectory, "YOUR_DIRECTORY", "ConvertedToPdfX4.pdf");
        pdfDocument.Save(outputPath);
        Console.WriteLine($"ЁЯТ╛ Converted PDF/XтАС4 saved at: {outputPath}");

        // -------------------------------------------------
        // 5я╕ПтГг Clean up (important for ASP.NET)
        // -------------------------------------------------
        signatureHandler.Dispose();
        pdfDocument.Dispose();
    }
}
```
