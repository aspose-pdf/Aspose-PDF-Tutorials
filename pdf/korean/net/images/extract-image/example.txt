//ExStart
//ExSummary: 이 코드는 PDF 문서에서 이미지를 추출하고 JPEG 파일로 저장하는 방법을 보여줍니다.
//ExStepSummary:0: 다음 코드에서 문서 디렉토리로 가는 경로를 초기화합니다.
//ExStepSummary:1: 다음 코드에서는 PDF 문서를 열면 그 문서를 통해 이미지를 추출할 수 있습니다.
//ExStepSummary:2: 다음 코드에서는 PDF 문서에서 특정 이미지를 추출합니다.
//ExStepSummary:3: 다음 코드는 추출된 이미지를 지정된 출력 파일로 저장합니다.
//ExStepSummary:4: 다음 코드는 원래 PDF 파일을 업데이트하고 새로운 경로로 저장합니다.
//ExStepImage:3:images/1.png
//ExStep:0-
// PDF 파일을 포함하는 디렉토리를 지정합니다
string dataDir = "Your Document Directory";

//ExStep:1-
// 이미지 추출을 위해 PDF 문서를 열
Document pdfDocument = new Document(dataDir + "ExtractImages.pdf");

//ExStep:2-
// PDF의 두 번째 페이지에서 첫 번째 이미지를 추출합니다
XImage xImage = pdfDocument.Pages[1].Resources.Images[1];

//ExStep:3-
// 추출된 이미지를 저장하기 위해 스트림을 만듭니다
FileStream outputImage = new FileStream(dataDir + "output.jpg", FileMode.Create);

// 출력 이미지를 저장
xImage.Save(outputImage, ImageFormat.Jpeg); 
// 리소스를 풀기 위해 파일 스트림을 닫습니다
outputImage.Close();

//ExStep:4-
// PDF의 새로운 버전의 경로를 업데이트합니다
dataDir = dataDir + "ExtractImages_out.pdf";

// 업데이트 된 PDF 파일을 저장
pdfDocument.Save(dataDir);
// 파일의 성공적인 추출 및 저장을 확인합니다
Console.WriteLine("\nImages extracted successfully.\nFile saved at " + dataDir);
//ExEnd
