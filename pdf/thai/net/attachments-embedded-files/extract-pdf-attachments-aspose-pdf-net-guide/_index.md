## คู่มือการใช้งาน
### การเปิดเอกสาร PDF
**ภาพรวม:** เปิดเอกสาร PDF ที่มีอยู่เพื่อเข้าถึงหน้าและคุณสมบัติของเอกสารนั้น

#### ขั้นตอนที่ 1: กำหนดเส้นทางไฟล์
```csharp
string dataDir = "YOUR_DOCUMENT_DIRECTORY" + "/GetAlltheAttachments.pdf";
```

#### ขั้นตอนที่ 2: เปิดเอกสาร
```csharp
document pdfDocument = new Document(dataDir);
// เอกสารนี้พร้อมให้ใช้งานเพื่อดำเนินการต่อไปได้แล้ว
```
### การเข้าถึงคอลเลกชันไฟล์ที่ฝังไว้
**ภาพรวม:** ดึงไฟล์ที่ฝังไว้ทั้งหมดภายใน PDF เพื่อเตรียมการแยกไฟล์

#### ขั้นตอนที่ 3: รับไฟล์ที่ฝังไว้
```csharp
EmbeddedFileCollection embeddedFiles = pdfDocument.EmbeddedFiles;
// ตอนนี้คุณมีสิทธิ์เข้าถึงการรวบรวมสิ่งที่แนบมาแล้ว
```
### การวนซ้ำและการแยกสิ่งที่แนบมา
**ภาพรวม:** ทำซ้ำผ่านสิ่งที่แนบมาแต่ละรายการ แยกรายละเอียด และบันทึกลงในดิสก์

#### ขั้นตอนที่ 4: วนผ่านสิ่งที่แนบมา
```csharp
int count = 1;
foreach (FileSpecification fileSpecification in embeddedFiles)
{
    string fileName = fileSpecification.Name;
    string description = fileSpecification.Description;
    string mimeType = fileSpecification.MIMEType;

    if (fileSpecification.Params != null)
    {
        // ดึงข้อมูลเมตาเพิ่มเติมหากมี
        string checksum = fileSpecification.Params.CheckSum;
        DateTime creationDate = fileSpecification.Params.CreationDate;
        DateTime modificationDate = fileSpecification.Params.ModDate;
        int size = fileSpecification.Params.Size;
    }

    // อ่านและบันทึกเนื้อหาที่แนบมา
    byte[] fileContent = new byte[fileSpecification.Contents.Length];
    fileSpecification.Contents.Read(fileContent, 0, fileContent.Length);
    string outputFilePath = dataDir + "/YOUR_OUTPUT_DIRECTORY/" + count + "_out.txt";
    File.WriteAllBytes(outputFilePath, fileContent);
    count++;
}
```
