## คู่มือการใช้งาน
มาแบ่งขั้นตอนการใช้งานออกเป็นขั้นตอนๆ กัน

### ขั้นตอนที่ 1: เตรียมสภาพแวดล้อมของคุณ
ตรวจสอบให้แน่ใจว่าไฟล์ TIFF ทั้งหมดถูกเก็บไว้ในไดเร็กทอรีที่คุณจะอ้างอิงในโค้ดของคุณ ตั้งค่าเส้นทางไฟล์:
```csharp
string dataDir = "YOUR_DOCUMENT_DIRECTORY";
```

### ขั้นตอนที่ 2: โหลดและแปลงไฟล์ TIFF เป็น PDF
แกนหลักของฟีเจอร์นี้เกี่ยวข้องกับการอ่านไฟล์ TIFF แต่ละไฟล์ แปลงให้เป็นอาร์เรย์ไบต์ จากนั้นเพิ่มเป็นรูปภาพบนหน้าใหม่ในเอกสาร PDF
```csharp
string[] files = Directory.GetFiles(dataDir, "*.tif");
Document doc = new Document();

foreach (string myFile in files)
{
    using (FileStream fs = new FileStream(myFile, FileMode.Open, FileAccess.Read))
    {
        byte[] tmpBytes = new byte[fs.Length];
        fs.Read(tmpBytes, 0, Convert.ToInt32(fs.Length));
        
        using (MemoryStream mystream = new MemoryStream(tmpBytes))
        {
            Bitmap b = new Bitmap(mystream);
            Page currpage = doc.Pages.Add();
            
            // กำหนดค่าระยะขอบและขนาดของหน้า
            currpage.PageInfo.Margin.Top = 5;
            currpage.PageInfo.Margin.Bottom = 5;
            currpage.PageInfo.Margin.Left = 5;
            currpage.PageInfo.Margin.Right = 5;
            currpage.PageInfo.Width = (b.Width / b.HorizontalResolution) * 72;
            currpage.PageInfo.Height = (b.Height / b.VerticalResolution) * 72;

            Image image1 = new Image();
            currpage.Paragraphs.Add(image1);
            
            // ตั้งค่า IsBlackWhite เพื่อปรับปรุงประสิทธิภาพ
            image1.IsBlackWhite = true;
            image1.ImageStream = mystream;
            image1.ImageScale = 0.95F; 
        }
    }
}
doc.Save("YOUR_DOCUMENT_DIRECTORY/PerformanceImprovement_out.pdf");
```

## การพิจารณาประสิทธิภาพ
- ใช้ `IsBlackWhite` คุณสมบัติในการปรับปรุงความเร็วในการเรนเดอร์และลดการใช้หน่วยความจำเมื่อไม่จำเป็นต้องใช้สี
