## คู่มือการใช้งาน

### การแปลง PDF เป็น HTML ด้วย URL รูปภาพที่กำหนดเอง

การแปลงเอกสาร PDF เป็น HTML พร้อมควบคุม URL ของรูปภาพนั้นต้องเข้าใจความสามารถของ Aspose.PDF และกำหนดค่าตัวเลือกที่เหมาะสม เราจะอธิบายขั้นตอนต่างๆ เหล่านี้ทีละขั้นตอน

#### ขั้นตอนที่ 1: โหลดเอกสาร
ขั้นแรก โหลดเอกสาร PDF ของคุณโดยใช้ `Document` ระดับ:

```csharp
// โหลดเอกสาร PDF ที่มีอยู่
document pdfDocument = new Document("input.pdf");
```

#### ขั้นตอนที่ 2: กำหนดค่าตัวเลือกการบันทึก
ตั้งค่า `HtmlSaveOptions` เพื่อระบุวิธีการจัดการรูปภาพระหว่างการแปลง สิ่งสำคัญคือการตั้งค่า `RasterImagesSavingMode` และการกำหนดกลยุทธ์การประหยัดทรัพยากรแบบกำหนดเอง:

```csharp
// ตั้งค่าตัวเลือกการบันทึก HTML
HtmlSaveOptions options = new HtmlSaveOptions();

// กำหนดโหมดการบันทึกภาพ
options.RasterImagesSavingMode = HtmlSaveOptions.RasterImagesSavingModes.AsExternalPngFilesReferencedViaSvg;

// กำหนดกลยุทธ์การประหยัดทรัพยากรแบบกำหนดเอง
customResourceSavingStrategy: new HtmlSaveOptions.ResourceSavingStrategy(Custom_processor_of_embedded_images);
```

#### ขั้นตอนที่ 3: นำกลยุทธ์การประหยัดทรัพยากรแบบกำหนดเองมาใช้
คุณสามารถปรับแต่งวิธีการบันทึกและอ้างอิงรูปภาพได้ที่นี่:

```csharp
private static string Custom_processor_of_embedded_images(SaveOptions.ResourceSavingInfo resourceSavingInfo)
{
    if (!(resourceSavingInfo is HtmlSaveOptions.HtmlImageSavingInfo))
    {
        resourceSavingInfo.CustomProcessingCancelled = true;
        return "";
    }

    // กำหนดไดเรกทอรีเอาท์พุตสำหรับรูปภาพ
    string dataDir = "your/data/directory/path";
    string outDir = Path.Combine(dataDir, @"output\files");
    string outPath = Path.Combine(outDir, Path.GetFileName(resourceSavingInfo.SupposedFileName));

    if (!Directory.Exists(outDir))
    {
        Directory.CreateDirectory(outDir);
    }

    // บันทึกภาพ
    using (BinaryReader reader = new BinaryReader(resourceSavingInfo.ContentStream))
    {
        System.IO.File.WriteAllBytes(outPath, reader.ReadBytes((int)resourceSavingInfo.ContentStream.Length));
    }

    // คืน URL ที่กำหนดเองเพื่ออ้างอิงรูปภาพในรูปแบบ HTML/SVG
    HtmlSaveOptions.HtmlImageSavingInfo imgInfo = (HtmlSaveOptions.HtmlImageSavingInfo)resourceSavingInfo;
    
    if (imgInfo.ParentType == HtmlSaveOptions.ImageParentTypes.SvgImage)
    {
        return "http://localhost:255/" + resourceSavingInfo.SuposedFileName;
    }
    else
    {
        return "http://localhost:GetImage/imageID=" + resourceSavingInfo.SuposedFileName;
    }
}
```

ฟังก์ชันนี้จะกำหนดวิธีบันทึกและอ้างอิงรูปภาพ ซึ่งช่วยให้คุณระบุ URL ที่กำหนดเองได้

#### ขั้นตอนที่ 4: ดำเนินการแปลง
สุดท้ายให้บันทึกเอกสารในรูปแบบ HTML โดยใช้ตัวเลือกที่คุณกำหนดค่าไว้:

```csharp
// บันทึก PDF เป็น HTML
document.Save("output.html", options);
```
