            // Set up optimization options

                ImageCompression   = ImageCompressionMode.JpegLossless,

                // ImageCompression = ImageCompressionMode.JpegLossy,

            // Save the new file
            doc.Save(outputPath);
        }

        Console.WriteLine($"‚úÖ Optimized PDF saved to: {outputPath}");
    }
}
